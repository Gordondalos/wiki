допустим у нас есть два конструктора один с двумя передаваемыми параметрами 
и второй с одним передаваемым параметром

можно запустить конструктор с одним параметром и в нем запустить конструктор с двумя параметрами
например

Myclass
{
	public Myclass(int a, int b)
	{
		Console.WriteLine("Этот с двумя);
	}
	
	public Myclass(string name)
		: this (300, 500)
	{
		Console.WriteLine("Этот с одним но он вызовит с двумя тоже);
	}


}

Program
{
	static void Main()
	{
		// в данном случае конструктор с одним параметром вызовит конструктор с двумя
		// параметрами со значениями по умолчанию 300 и 500
		Myclass name = new Myclass("Вася");
	}
}
